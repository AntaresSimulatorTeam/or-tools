213 // The following code defines needed classes for the KnapsackGenericSolver
214 // class which is the entry point to extend knapsack with new constraints such
215 // as conflicts between items.
216 //
217 // Constraints are enforced using KnapsackPropagator objects, in the current
218 // code there is one propagator per dimension (KnapsackCapacityPropagator).
219 // One of those propagators, named master propagator, is used to guide the
220 // search, i.e. decides which item should be assigned next.
221 // Roughly speaking the search algorithm is:
222 // - While not optimal
223 // - Select next search node to expand
224 // - Select next item_i to assign (using master propagator)
225 // - Generate a new search node where item_i is in the knapsack
226 // - Check validity of this new partial solution (using propagators)
227 // - If valid, add this new search node to the search
228 // - Generate a new search node where item_i is not in the knapsack
229 // - Check validity of this new partial solution (using propagators)
230 // - If valid, add this new search node to the search
231 //
232 // TODO(user): Add a new propagator class for conflict constraint.
233 // TODO(user): Add a new propagator class used as a guide when the problem has
234 // several dimensions.

246 // ----- KnapsackItem -----
247 // KnapsackItem is a small struct to pair an item weight with its
248 // corresponding profit.
249 // The aim of the knapsack problem is to pack as many valuable items as
250 // possible. A straight forward heuristic is to take those with the greatest
251 // profit-per-unit-weight. This ratio is called efficiency in this
252 // implementation. So items will be grouped in vectors, and sorted by
253 // decreasing efficiency.
254 // Note that profits are duplicated for each dimension. This is done to
255 // simplify the code, especially the GetEfficiency method and vector sorting.
256 // As there usually are only few dimensions, the overhead should not be an
257 // issue.
258 struct KnapsackItem {
259  KnapsackItem(int _id, int64 _weight, int64 _profit)
260  : id(_id), weight(_weight), profit(_profit) {}
261  double GetEfficiency(int64 profit_max) const {
262  return (weight > 0)
263  ? static_cast<double>(profit) / static_cast<double>(weight)
264  : static_cast<double>(profit_max);
265  }
266 
267  // The 'id' field is used to retrieve the initial item in order to
268  // communicate with other propagators and state.
269  const int id;
270  const int64 weight;
271  const int64 profit;
272 };

321 // ----- KnapsackSearchPath -----
322 // KnapsackSearchPath is a small class used to represent the path between a
323 // node to another node in the search tree.
324 // As the solution state is not stored for each search node, the state should
325 // be rebuilt at each node. One simple solution is to apply all decisions
326 // between the node 'to' and the root. This can be computed in
327 // O(number_of_items).
328 //
329 // However, it is possible to achieve better average complexity. Two
330 // consecutively explored nodes are usually close enough (i.e., much less than
331 // number_of_items) to benefit from an incremental update from the node
332 // 'from' to the node 'to'.
333 //
334 // The 'via' field is the common parent of 'from' field and 'to' field.
335 // So the state can be built by reverting all decisions from 'from' to 'via'
336 // and then applying all decisions from 'via' to 'to'.
337 class KnapsackSearchPath {
338  public:
339  KnapsackSearchPath(const KnapsackSearchNode& from,
340  const KnapsackSearchNode& to);
341  void Init();
342  const KnapsackSearchNode& from() const { return from_; }
343  const KnapsackSearchNode& via() const { return *via_; }
344  const KnapsackSearchNode& to() const { return to_; }
345  const KnapsackSearchNode* MoveUpToDepth(const KnapsackSearchNode& node,
346  int depth) const;
347 
348  private:
349  const KnapsackSearchNode& from_;
350  const KnapsackSearchNode* via_; // Computed in 'Init'.
351  const KnapsackSearchNode& to_;
352 
353  DISALLOW_COPY_AND_ASSIGN(KnapsackSearchPath);
354 };

458 // ----- KnapsackCapacityPropagator -----
459 // KnapsackCapacityPropagator is a KnapsackPropagator used to enforce
460 // a capacity constraint.
461 // As a KnapsackPropagator is supposed to compute profit lower and upper
462 // bounds, and get the next item to select, it can be seen as a 0-1 Knapsack
463 // solver. The most efficient way to compute the upper bound is to iterate on
464 // items in profit-per-unit-weight decreasing order. The break item is
465 // commonly defined as the first item for which there is not enough remaining
466 // capacity. Selecting this break item as the next-item-to-assign usually
467 // gives the best results (see Greenberg & Hegerich).
468 //
469 // This is exactly what is implemented in this class.
470 //
471 // When there is only one propagator, it is possible to compute a better
472 // profit lower bound almost for free. During the scan to find the
473 // break element all unbound items are added just as if they were part of
474 // the current solution. This is used in both ComputeProfitBounds and
475 // CopyCurrentSolutionPropagator.
476 // For incrementality reasons, the ith item should be accessible in O(1). That's
477 // the reason why the item vector has to be duplicated 'sorted_items_'.
478 class KnapsackCapacityPropagator : public KnapsackPropagator {
479  public:
480  KnapsackCapacityPropagator(const KnapsackState& state, int64 capacity);
481  ~KnapsackCapacityPropagator() override;
482  void ComputeProfitBounds() override;
483  int GetNextItemId() const override { return break_item_id_; }
484 
485  protected:
486  // Initializes KnapsackCapacityPropagator (e.g., sort items in decreasing
487  // order).
488  void InitPropagator() override;
489  // Updates internal data structure incrementally (i.e., 'consumed_capacity_')
490  // to avoid a O(number_of_items) scan.
491  bool UpdatePropagator(bool revert,
492  const KnapsackAssignment& assignment) override;
493  void CopyCurrentStateToSolutionPropagator(
494  std::vector<bool>* solution) const override;
495 
496  private:
497  // An obvious additional profit upper bound corresponds to the linear
498  // relaxation: remaining_capacity * efficiency of the break item.
499  // It is possible to do better in O(1), using Martello-Toth bound U2.
500  // The main idea is to enforce integrality constraint on the break item,
501  // ie. either the break item is part of the solution, either it is not.
502  // So basically the linear relaxation is done on the item before the break
503  // item, or the one after the break item.
504  // This is what GetAdditionalProfit method implements.
505  int64 GetAdditionalProfit(int64 remaining_capacity, int break_item_id) const;
506 
507  const int64 capacity_;
508  int64 consumed_capacity_;
509  int break_item_id_;
510  std::vector<KnapsackItemPtr> sorted_items_;
511  int64 profit_max_;
512 
513  DISALLOW_COPY_AND_ASSIGN(KnapsackCapacityPropagator);
514 };
